97 template<typename T = std::string>
98 inline std::optional<T> get (std::string_view key) {
99 const auto jsonVal { _getJsonValue (key) };
100 if (jsonVal.has_value()) {
101 if constexpr (std::is_same_v<T, bool>) {
102 return static_cast<T> (jsonVal.value().get().get<bool>());
103 }
104 else if constexpr (std::is_integral_v<T>) {
105 return static_cast<T> (jsonVal.value().get().get<int64_t>());
106 }
107 else if constexpr (std::is_floating_point_v<T>) {
108 return static_cast<T> (jsonVal.value().get().get<double>());
109 }
110 else if constexpr (
111 (std::is_same_v<T, std::vector<std::string>>) ||
112 (std::is_same_v<T, std::vector<double>>) ||
113 (std::is_same_v<T, std::vector<int64_t>>) ||
114 (std::is_same_v<T, std::vector<bool>>)
115 ) {
116 T result;
117 for (const auto &jv: jsonVal.value().get().asArray())
118 result.push_back (jv.get<typename T::value_type>());
119 return result;
120 }
121 else if constexpr (
122 (std::is_same_v<T, std::vector<int32_t>>) ||
123 (std::is_same_v<T, std::vector<int16_t>>) ||
124 (std::is_same_v<T, std::vector<int8_t>>) ||
125 (std::is_same_v<T, std::vector<uint64_t>>) ||
126 (std::is_same_v<T, std::vector<uint32_t>>) ||
127 (std::is_same_v<T, std::vector<uint16_t>>) ||
128 (std::is_same_v<T, std::vector<uint8_t>>)
129 ) {
130 T result;
131 for (const auto &jv: jsonVal.value().get().asArray())
132 result.push_back (static_cast<typename T::value_type> (jv.get<int64_t>()));
133 return result;
134 }
135 else if constexpr (std::is_same_v<T, std::vector<float>>) {
136 T result;
137 for (const auto &jv: jsonVal.value().get().asArray())
138 result.push_back (static_cast<typename T::value_type> (jv.get<double>()));
139 return result;
140 }
141 else {
142 return jsonVal.value().get().get<T>();
143 }
144 }
145
146 return std::nullopt;
147 }
