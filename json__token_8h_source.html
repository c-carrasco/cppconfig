159 case JsonTokenId::kValueFloatPoint: os << "kValueFloatPoint[" << std::get<double> (obj._value) << "]"; break;

161 case JsonTokenId::kValueBoolean: os << "ValueBoolean[" << std::get<bool> (obj._value) << "]"; break;
162 case JsonTokenId::kValueNull: os << "ValueNull"; break;
163 case JsonTokenId::kError: os << "Error"; break;
